About the Collection

Have you ever wanted to relive the trading card experience, but feel you lack the time and effort to physically collect the cards? Worried that you could lose your most valuable collectibles, or risk trading into fakes?

We get it, and that’s why we created the NFT card shop, to make trading easier and hassle-free for the end user. We’re a digital trading card outlet that ties virtual our original trading card NFTs to physical graded cards of the same design.

It’s all done using a complex smart contract, with the value of the digital cards directly linked to a physical card that we store securely on your behalf. This means that when you trade, you don’t have to think about logistics or authenticity - because we’ve got it covered. All the fun, none of the fuss.

The Contract

This is our contract. Built on top of ERC721, the industry standard for serious non-fungible tokens. This is the same set of standards used in the creation of the BAYC collection. The assets are stored on IPFS, so that your NFT will remain with you no matter what happens to us.

Our Contract on Etherscan 
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.10;

import "@openzeppelin/contracts/token/ERC721/ERC721.sol";
import "@openzeppelin/contracts/token/ERC721/extensions/ERC721URIStorage.sol";
import "@openzeppelin/contracts/access/Ownable.sol";
import "@openzeppelin/contracts/utils/Counters.sol";

contract NFTCardShop is ERC721, ERC721URIStorage, Ownable {
    using Counters for Counters.Counter;

    Counters.Counter private _tokenIdCounter;

    uint256 public tokenTransferUnlockRequestWaitPeriod;

    uint256 public constant MAXIMUM_TOKEN_TRANSFER_UNLOCK_REQUEST_WAITING_PERIOD = 40 days;

    event TokenTransferUnlockRequest(uint256 tokenId, uint256 tokenTransferUnlockRequestDate);

    // tokenId => token is in vault, verified, and graded
    mapping(uint256 => bool) public tokenIsInVaultAndGraded;

    // tokenId => last token transfer date as seconds since unix epoch
    mapping(uint256 => uint256) public tokenIdToTokenTransferUnlockRequestDate;

    bool public cardShopRetainsTransferControl;

    constructor() ERC721("NFTCardShop", "CaSh") {
        cardShopRetainsTransferControl = true;
        tokenTransferUnlockRequestWaitPeriod = MAXIMUM_TOKEN_TRANSFER_UNLOCK_REQUEST_WAITING_PERIOD;
    }

    function _baseURI() internal pure override returns (string memory) {
        return "https://gateway.pinata.cloud/ipfs/";
    }

    function safeMint(address to, string memory uri) public onlyOwner {
        uint256 tokenId = _tokenIdCounter.current();
        _tokenIdCounter.increment();
        _safeMint(to, tokenId);
        _setTokenURI(tokenId, uri);
        tokenIsInVaultAndGraded[tokenId] = true;
        assert(tokenIdToTokenTransferUnlockRequestDate[tokenId] == 0);
    }

    function _beforeTokenTransfer(
        address from, 
        address to, 
        uint256 tokenId
    ) internal override(ERC721) {
        require(tokenIsInVaultAndGraded[tokenId] || !cardShopRetainsTransferControl);
    }

    // true => token was transfer unlocked
    // false => token was not transfer unlocked
    // NOTE: The idea here is that we are giving the vault an amount of time to
    // recieve, verify, and grade a card coming back into the vault. However,
    // in the event that the vault stops operating for any reason, we want to provide
    // a mechanism by which holders of an NFT from this collection will still be able to
    // trade. This amount of time is defined by the contract member variable `transferTokenUnlockRequestWaitPeriod`.
    // If the vault has stopped operating, then it will not call the `removeTokenTransferUnlockRequest(uint256 tokenId)` function.
    // As such, after that amount of time has passed, the card (aka token, NFT) holder will be able to call this function again
    // which will allow the holder to conduct transactions as if the card was in the vault.
    // NOTE 2: We do this to protect the collection from even us and give the community the ability to adapt to
    // a situation where the vault is no longer operational. This mechanism makes our collection future-proof.
    function attemptTokenTransferUnlock(uint256 tokenId) public returns(bool) {
        require(!tokenIsInVaultAndGraded[tokenId], "NFTCardShop: No need to request unlock. Cards that are in the vault can be transferred at will.");
        if (tokenIdToTokenTransferUnlockRequestDate[tokenId] == 0) {
            tokenIdToTokenTransferUnlockRequestDate[tokenId] = block.timestamp;
            emit TokenTransferUnlockRequest(tokenId, tokenIdToTokenTransferUnlockRequestDate[tokenId]);
            return false;
        } else if ((tokenIdToTokenTransferUnlockRequestDate[tokenId] + tokenTransferUnlockRequestWaitPeriod) >= block.timestamp) {
            tokenIsInVaultAndGraded[tokenId] = true;
            return true;
        }
        return false;
    }

    function cardRemovedFromVault(uint256 tokenId) public onlyOwner {
        require(tokenIsInVaultAndGraded[tokenId], "NFTCardShop: Card cannot be removed from the vault unless it is currently in the vault.");
        tokenIsInVaultAndGraded[tokenId] = false;
        // remove vault checkmark (using setTokenURI)
        //setTokenURI(tokenId, /* URI without checkmark */);
    }

    function cardReturnedToVault(uint256 tokenId) public onlyOwner {
        require(!tokenIsInVaultAndGraded[tokenId], "NFTCardShop: Card cannot be returned to the vault if it is already in the vault.");
        removeTokenTransferUnlockRequest(tokenId);
        tokenIsInVaultAndGraded[tokenId] = true;
        // add vault checkmark to NFT (using setTokenURI)
        //setTokenURI(tokenId, /* URI with checkmark */);
    }

    function removeTokenTransferUnlockRequest(uint256 tokenId) public onlyOwner {
        tokenIdToTokenTransferUnlockRequestDate[tokenId] = 0;
    }

    // TODO: Implement a voting mechanism for our Token holders.
    // This will allow the community to decide whether or not they
    // want the trading security services of the vault
    function toggleTransferControl() public onlyOwner {
        cardShopRetainsTransferControl = !cardShopRetainsTransferControl;
    }

    function setTokenTransferUnlockRequestWaitPeriod(uint256 amountOfTime) public onlyOwner {
        require(amountOfTime <= MAXIMUM_TOKEN_TRANSFER_UNLOCK_REQUEST_WAITING_PERIOD);
        tokenTransferUnlockRequestWaitPeriod = amountOfTime;
    }

    // will be used to update the NFT with a "Card Secured In Vault" checkmark
    function setTokenURI(uint256 tokenId, string memory _tokenURI) private onlyOwner {
        super._setTokenURI(tokenId, _tokenURI);
    }

    // ==== Required Overrides (required by solidity) ====

    function _burn(uint256 tokenId) internal override(ERC721, ERC721URIStorage) {
        super._burn(tokenId);
    }

    function tokenURI(uint256 tokenId) public view override(ERC721, ERC721URIStorage) returns (string memory) {
        return super.tokenURI(tokenId);
    }
}
